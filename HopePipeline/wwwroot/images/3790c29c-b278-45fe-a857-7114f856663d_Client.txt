import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.io.PrintWriter;
import java.net.Socket;
import java.net.UnknownHostException;

public class Client extends Thread{
 public String selection;
	public Integer loopNum;
	public String address;
 public Socket socket;
	public Client(String meh, String ip, int numTime) throws UnknownHostException, IOException {
	loopNum = numTime;
        selection = meh;
        address = ip;
        
        
    }
	 public void run() {
		
		

		long startTime = System.currentTimeMillis();
		 PrintWriter writer;
		 while(true) {
		try {
			 socket = new Socket(address, 3001);
			writer = new PrintWriter(socket.getOutputStream());
			writer.println(selection);
			 writer.flush();
			 
			 String line = null;
			 String totalRecieve = null;
			 InputStream input = socket.getInputStream();
			   BufferedReader reader = new BufferedReader(new InputStreamReader(input));
			   do{
				    line = reader.readLine();
				    totalRecieve+=line;// reads a line of text
			   } while(line!=null);
			long endTime = System.currentTimeMillis();
			   long duration = (endTime - startTime);
			   
			   //String totalTime1 = Long.toString(totalTime);
			   System.out.println(duration);
			//  System.out.println("message recieved from server " + totalRecieve);
			break;
		} catch (IOException e) {
			// TODO Auto-generated catch block
			System.out.println("error");
		}
		 
		 }
	}	
    
}
